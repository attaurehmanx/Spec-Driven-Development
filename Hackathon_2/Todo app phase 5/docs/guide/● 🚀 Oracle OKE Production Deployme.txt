● 🚀 Oracle OKE Production Deployment Guide

  Prerequisites

  1. Oracle Cloud Account - Sign up at https://cloud.oracle.com
  2. Install OCI CLI - bash -c "$(curl -L
  https://raw.githubusercontent.com/oracle/oci-cli/master/scripts/install/install.sh)"
  3. Configure OCI CLI - Run oci setup config
  4. Install kubectl and helm (already done)
  5. Domain name - Purchase a domain for your application

  ---
  Step 1: Create OKE Cluster

  Via OCI Console:
  1. Log in to Oracle Cloud Console
  2. Menu → Developer Services → Kubernetes Clusters (OKE)
  3. Click "Create Cluster" → Choose "Quick Create"
  4. Configure:
    - Name: task-management-prod
    - Kubernetes Version: Latest stable (1.28+)
    - Node Shape: VM.Standard.E4.Flex (2 OCPUs, 16GB RAM)
    - Number of nodes: 3
  5. Click "Create Cluster" (takes 10-15 minutes)

  ---
  Step 2: Configure kubectl for OKE

  1. Go to your cluster in OCI Console
  2. Click "Access Cluster"
  3. Copy and run the kubeconfig command:
  oci ce cluster create-kubeconfig --cluster-id <cluster-ocid> --file $HOME/.kube/config --region <region>
  4. Verify: kubectl get nodes

  ---
  Step 3: Push Images to Oracle Container Registry (OCIR)

  Create Auth Token:
  1. OCI Console → User Settings → Auth Tokens
  2. Generate Token → Save it (you won't see it again!)

  Login and Push:
  # Login to OCIR
  docker login <region-key>.ocir.io
  # Username: <tenancy-namespace>/<oci-username>
  # Password: <auth-token>

  # Tag images
  docker tag task-management-backend:latest <region>.ocir.io/<tenancy>/task-management/backend:latest
  docker tag task-management-frontend:latest <region>.ocir.io/<tenancy>/task-management/frontend:latest

  # Push images
  docker push <region>.ocir.io/<tenancy>/task-management/backend:latest
  docker push <region>.ocir.io/<tenancy>/task-management/frontend:latest

  ---
  Step 4: Create Kubernetes Secrets

  # OCIR pull secret
  kubectl create secret docker-registry ocir-secret \
    --docker-server=<region>.ocir.io \
    --docker-username='<tenancy-namespace>/<username>' \
    --docker-password='<auth-token>'

  # Application secrets
  kubectl create secret generic api-secrets \
    --from-literal=gemini-api-key=<your-key> \
    --from-literal=auth-secret=<random-string-64-chars> \
    --from-literal=secret-key=<random-string-64-chars> \
    --from-literal=database-url=<neon-postgres-url>

  ---
  Step 5: Install Nginx Ingress Controller

  helm repo add ingress-nginx https://kubernetes.github.io/ingress-nginx
  helm repo update

  helm install nginx-ingress ingress-nginx/ingress-nginx \
    --namespace ingress-nginx --create-namespace \
    --set controller.service.type=LoadBalancer

  # Get external IP (save this for DNS)
  kubectl get svc -n ingress-nginx

  ---
  Step 6: Configure DNS

  1. Get LoadBalancer external IP from Step 5
  2. Go to your domain registrar (GoDaddy, Namecheap, etc.)
  3. Add DNS A records:
    - taskmanagement.example.com → <LoadBalancer-IP>
    - api.taskmanagement.example.com → <LoadBalancer-IP>
  4. Wait for DNS propagation (5-30 minutes)

  ---
  Step 7: Install cert-manager for SSL

  # Install cert-manager
  kubectl apply -f https://github.com/cert-manager/cert-manager/releases/download/v1.14.0/cert-manager.yaml

  # Wait for ready
  kubectl wait --for=condition=ready pod -l app.kubernetes.io/instance=cert-manager -n cert-manager --timeout=300s

  # Create Let's Encrypt issuer
  cat <<EOF | kubectl apply -f -
  apiVersion: cert-manager.io/v1
  kind: ClusterIssuer
  metadata:
    name: letsencrypt-prod
  spec:
    acme:
      server: https://acme-v02.api.letsencrypt.org/directory
      email: your-email@example.com
      privateKeySecretRef:
        name: letsencrypt-prod
      solvers:
      - http01:
          ingress:
            class: nginx
  EOF

  ---
  Step 8: Install Dapr (High Availability)

  dapr init -k --enable-ha=true --enable-mtls=true

  # Verify
  dapr status -k

  ---
  Step 9: Install Redpanda (Kafka)

  helm repo add redpanda https://charts.redpanda.com
  helm repo update

  helm install redpanda redpanda/redpanda \
    --set statefulset.replicas=3 \
    --set resources.memory.container.max=4Gi \
    --set storage.persistentVolume.enabled=true \
    --set storage.persistentVolume.size=20Gi

  # Create topic
  kubectl exec redpanda-0 -- rpk topic create tasks.reminder --partitions 3 --replicas 3

  ---
  Step 10: Deploy Dapr Components

  kubectl apply -f helm/templates/dapr-components/kafka-pubsub.yaml

  ---
  Step 11: Update Deployment Files

  Edit helm/templates/backend-deployment.yaml:
  - Change image: to OCIR path: <region>.ocir.io/<tenancy>/task-management/backend:latest
  - Change imagePullPolicy: Never to Always
  - Add imagePullSecrets: - name: ocir-secret
  - Change replicas: 1 to replicas: 3
  - Add Ingress with your domain

  Edit helm/templates/frontend-deployment.yaml:
  - Change image: to OCIR path: <region>.ocir.io/<tenancy>/task-management/frontend:latest
  - Change imagePullPolicy: Never to Always
  - Add imagePullSecrets: - name: ocir-secret
  - Change replicas: 1 to replicas: 3
  - Add Ingress with your domain

  ---
  Step 12: Deploy Applications

  # Deploy backend
  kubectl apply -f helm/templates/backend-deployment.yaml

  # Deploy frontend
  kubectl apply -f helm/templates/frontend-deployment.yaml

  # Wait for deployments
  kubectl wait --for=condition=available deployment/task-management-backend --timeout=300s
  kubectl wait --for=condition=available deployment/task-management-frontend --timeout=300s

  ---
  Step 13: Verify Deployment

  # Check pods
  kubectl get pods -l app=task-management

  # Check services
  kubectl get svc -l app=task-management

  # Check ingress
  kubectl get ingress

  # Check certificates
  kubectl get certificate

  ---
  Step 14: Access Your Application

  1. Wait for SSL certificates to be issued (2-5 minutes)
  2. Open browser: https://taskmanagement.example.com
  3. Backend API: https://api.taskmanagement.example.com

  ---
  Step 15: Set Up Monitoring (Optional)

  # Install Prometheus
  helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
  helm install prometheus prometheus-community/kube-prometheus-stack

  # Install Grafana (included in above)
  # Access: kubectl port-forward svc/prometheus-grafana 3000:80

  ---
  Production Checklist

  - ✅ 3+ replicas for HA
  - ✅ SSL/TLS certificates
  - ✅ Resource limits set
  - ✅ Health checks configured
  - ✅ Persistent storage for Redpanda
  - ✅ Secrets properly managed
  - ✅ Monitoring enabled
  - ✅ Backups configured
  - ✅ DNS configured
  - ✅ Dapr mTLS enabled

  ---
  Cost Estimate (Oracle OKE)

  - OKE Cluster: Free (control plane)
  - 3 Worker Nodes (VM.Standard.E4.Flex): ~$150/month
  - LoadBalancer: ~$10/month
  - Block Storage (60GB): ~$6/month
  - Total: ~$166/month

  ---
  Troubleshooting

  Pods not starting:
  kubectl describe pod <pod-name>
  kubectl logs <pod-name>

  SSL certificate not issued:
  kubectl describe certificate <cert-name>
  kubectl logs -n cert-manager deployment/cert-manager

  Can't pull images:
  kubectl get secret ocir-secret -o yaml
  # Verify credentials are correct

  That's the complete production deployment guide! Follow these steps in order for a successful Oracle OKE deployment.
